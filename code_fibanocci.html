Fibanocci


def fibonacci(n):
a = 0
b = 1
if n == 1:
    return b
else:
    for i in range(2,n):
        c = a + b
        a = b
        b = c
    return b

Fibanocci récursif


def Fibonacci(n):
if n <= 1:
    return 1
else:
    return Fibonacci(n-1) + Fibonacci(n-2)
